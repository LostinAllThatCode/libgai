Detailed Description

Requests a window for all currently supported platforms.

Attention
ONLY WINDOWS SUPPORT FOR NOW
Renderer Linker Dependencies (win32) Linker Dependencies (linux)
Default user32.lib Not implemented yet!
OpenGL user32.lib winmm.lib gdi32.lib opengl32.lib Not implemented yet!
Note
When using the msvc compiler, #pragma comment( lib, "xxx" ) commands will automatically link to these libs.
Otherwise you have to tell the linker which libs it has to link to.

Introduction

To use this api you need to specifiy some macro which define the window initialization process.

Do this:

#define GAIXW_IMPLEMENTATION

before you include this file in one C or C++ file to create the implementation.

Examples

GDI Example

#define GAIXW_IMPLEMENTATION
#include <gai_xwindow.h>
#include <stdio.h>
LRESULT CALLBACK
MyWindowMessageHandler(HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
{
gaixw_context *window = (gaixw_context *) GetWindowLongPtr(hWnd, GWLP_USERDATA);
if (window)
{
switch (uMsg)
{
case WM_PAINT:
{
RECT r = { 0, 0, 120, 50 };
DrawText(window->platform.win32.hdc, "Hallo", 5, &r, DT_INTERNAL | DT_NOCLIP);
} break;
}
}
return gaixw_WindowProc(hWnd, uMsg, wParam, lParam);
}
int main(int argc, char** argv)
{
int retval = 0;
gaixw_context window = {};
retval = gaixw_Init(&window, "gdi example", 640, 480);
if(retval != 1) return retval;
printf("Renderer: %u\n", window.renderer.type);
SetWindowLongPtr(window.platform.win32.hwnd, GWLP_WNDPROC, (LONG_PTR) MyWindowMessageHandler);
gaixw_Update(&window);
gaixw_Deinit(&window);
return 0;
}

OpenGL Example

#define GAIXW_OPENGL
#define GAIXW_IMPLEMENTATION
#include <gai_xwindow.h>
#include <stdio.h>
int main(int argc, char** argv)
{
int retval = 0;
gaixw_context window = {};
retval = gaixw_Init(&window, "opengl example", 640, 480);
if (retval != 1) return retval;
printf("Vendor: %s, Version: %s, Renderer: %s, Shading Language Version: %s, OpenGL Context: %p\n",
window.renderer.interface.opengl.vendor, window.renderer.interface.opengl.version, window.renderer.interface.opengl.renderer,
window.renderer.interface.opengl.shading_language_version, window.renderer.interface.opengl.context);
gaixw_SetVerticalSync(&window, 1);
glClearColor(0.f, 0.f, 0.f, 1.f);
for (;;)
{
gaixw_Update(&window);
if (!window.is_running) break;
glClear(GL_COLOR_BUFFER_BIT);
glBegin(GL_QUADS);
{
glColor4f(1.f, 0.f, 0.0f, 1.f);
glVertex2f(-.5f, -.5f);
glColor4f(0.f, 1.f, 0.0f, 1.f);
glVertex2f(-.5f, .5f);
glColor4f(1.f, 1.f, 1.0f, 1.f);
glVertex2f(.5f, .5f);
glColor4f(0.f, 0.f, 1.f, 1.f);
glVertex2f(.5f, -.5f);
}
glEnd();
glViewport(0, 0, window.info.width, window.info.height);
gaixw_SwapBuffers(&window);
}
gaixw_Deinit(&window);
return 0;
}

If you want to get a modern opengl context you have to specify these macros as well

#define GAIXW_OPENGL
#define GAIXW_OPENGL_MAJOR 3
#define GAIXW_OPENGL_MINOR 2
#define GAIXW_IMPLEMENTATION
#include <gai_xwindow.h>
