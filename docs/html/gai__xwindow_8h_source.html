630 gaixw_WGLIsSupported(const char *extension)
631 {
632  const size_t extlen = strlen(extension);
633  const char *supported = 0;
634  void* wglGetExtString = wglGetProcAddress("wglGetExtensionsStringARB");
635  if (wglGetExtString)
636  {
637  supported = ((char*(__stdcall*)(HDC)) wglGetExtString)(wglGetCurrentDC());
638  }
639 
640  if (supported)
641  {
642  for (const char* p = supported; ; p++)
643  {
644  p = strstr(p, extension);
645  if (!p) return false;
646  if ((p == supported || p[-1] == ' ') && (p[extlen] == '\0' || p[extlen] == ' ')) return true;
647  }
648  }
649  return false;

921 LRESULT CALLBACK
922 gaixw_WindowProc(HWND hWnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
923 {
924  gaixw_context *window = (gaixw_context *) GetWindowLongPtr(hWnd, GWLP_USERDATA);
925  if (window)
926  {
927  switch (uMsg)
928  {
929  case WM_SIZE:
930  {
931  window->info.width = LOWORD(lParam);
932  window->info.height = HIWORD(lParam);
933  switch (wParam)
934  {
935  case SIZE_RESTORED:
936  {
937  if (window->renderer.attributes & gaixwFlagsMinimized) window->renderer.attributes = window->renderer.attributes & ~gaixwFlagsMinimized;
938  if (window->renderer.attributes & gaixwFlagsMaximized) window->renderer.attributes = window->renderer.attributes & ~gaixwFlagsMaximized;
939  } break;
940  case SIZE_MAXIMIZED:
941  {
942  if (window->renderer.attributes & gaixwFlagsMinimized) window->renderer.attributes = window->renderer.attributes & ~gaixwFlagsMinimized;
943  window->renderer.attributes |= gaixwFlagsMaximized;
944  } break;
945  case SIZE_MINIMIZED:
946  {
947  if (window->renderer.attributes & gaixwFlagsMaximized) window->renderer.attributes = window->renderer.attributes & ~gaixwFlagsMaximized;
948  window->renderer.attributes |= gaixwFlagsMinimized;
949  } break;
950  }
951  return 0;
952  }
953  case WM_MOVE: { window->info.x = (int)(short) LOWORD(lParam); window->info.y = (int)(short) HIWORD(lParam); return 0; }
954  case WM_KEYDOWN: { window->input.keys_history[wParam] = window->input.keys[wParam]; window->input.keys[wParam] = 1; return 0; }
955  case WM_KEYUP: { window->input.keys_history[wParam] = window->input.keys[wParam]; window->input.keys[wParam] = 0; return 0; }
956  case WM_MOUSEMOVE: { POINTS p = MAKEPOINTS(lParam); window->input.x = p.x; window->input.y = p.y; return 0; }
957  case WM_LBUTTONDOWN: { window->input.mouse_history[0] = window->input.mouse[0]; window->input.mouse[0] = 1; return 0; }
958  case WM_LBUTTONUP: { window->input.mouse_history[0] = window->input.mouse[0]; window->input.mouse[0] = 0; return 0; }
959  case WM_MBUTTONDOWN: { window->input.mouse_history[1] = window->input.mouse[1]; window->input.mouse[1] = 1; return 0; }
960  case WM_MBUTTONUP: { window->input.mouse_history[1] = window->input.mouse[1]; window->input.mouse[1] = 0; return 0; }
961  case WM_RBUTTONDOWN: { window->input.mouse_history[2] = window->input.mouse[2]; window->input.mouse[2] = 1; return 0; }
962  case WM_RBUTTONUP: { window->input.mouse_history[2] = window->input.mouse[2]; window->input.mouse[2] = 0; return 0; }
963  case WM_MOUSEWHEEL: { window->input.dwheel = GET_WHEEL_DELTA_WPARAM(wParam) / WHEEL_DELTA; window->input.wheel += window->input.dwheel; return 0; }
964  case WM_DESTROY:
965  {
966  #if 0
967  _gaiXWindowDeInit(window);
968  #ifdef GAIXW_OPENGL
969  if (window->renderer.info.opengl.is_modern)
970  {
971  HGLRC ctx = wglGetCurrentContext();
972  if (ctx)
973  {
974  wglMakeCurrent(0, 0);
975  wglDeleteContext(ctx);
976  }
977  }
978  #endif
979  PostQuitMessage(0);
980  #endif
981  window->is_running = false;
982  return 0;
983  } break;
984  }
985  }
986  return DefWindowProc(hWnd, uMsg, wParam, lParam);

1059  static LARGE_INTEGER _ltframetime;
1060  static int fps;
1061  static int frames;
1062  static long double frametime;
1063 
1064  static unsigned int initialized;
1065  if (!initialized)
1066  {
1067  QueryPerformanceFrequency(&_frequency);
1068  QueryPerformanceCounter(&_ltframetime);
1069  initialized = true;
1070  }
1071 
1072  LARGE_INTEGER now, elapsed;
1073  QueryPerformanceCounter(&now);
1074  elapsed.QuadPart = now.QuadPart - _ltframetime.QuadPart;
1075  elapsed.QuadPart *= 1000000;
1076  elapsed.QuadPart /= _frequency.QuadPart;

1079  window->dt.millis = (float)elapsed.QuadPart / 1000.f;
1080  window->dt.seconds = (float)elapsed.QuadPart / 1000000.f;
1081  _ltframetime = now;
1082 
1083  int old_x = window->input.x;
1084  int old_y = window->input.y;
1085  window->input.dwheel = 0;
1086 
1087  while ( PeekMessageA(&msg, 0, 0, 0, PM_REMOVE) )
1088  {
1089  TranslateMessage(&msg);
1090  DispatchMessage(&msg);
1091  if (msg.message == WM_QUIT)
1092  {
1093  window->is_running = false;
1094  break;
1095  }
1096  }
1097 
1098  window->input.dx = window->input.x - old_x;
1099  window->input.dy = window->input.y - old_y;
1100 
1101  if (frametime >= 1.0f)
1102  {
1103  window->info.fps = frames;
1104  frames = 0;
1105  frametime = 0;
1106  }
1107  else
1108  {
1109  frametime += window->dt.seconds;
1110  frames++;
1111  }
1112 
1113  #else
1114 
1115  while (GetMessage(&msg, 0, 0, 0) != 0)
1116  {
1117  TranslateMessage(&msg);
1118  DispatchMessage(&msg);

1120  }
1121 
1122  #endif
1123

1125 }
